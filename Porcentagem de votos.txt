val = input ('Digite o número de votos válidos: ')
val = int(val)
bra = input ('Digite o número de votos brancos: ')
bra = int(bra)
nul = input ('Digite o número de votos nulos: ')
nul = int(nul)
porval = 100*(val/(val+bra+nul))
porbra = 100*(bra/(val+bra+nul))
pornul = 100*(nul/(val+bra+nul))
print('A porcentagem de votos válidos é:', porval)
print('A porcentagem de votos brancos é:', porbra)
print('A porcentagem de votos nulos é:', pornul)
